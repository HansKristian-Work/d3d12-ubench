file(MAKE_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR})

include(FindPythonInterp)
execute_process(COMMAND ${PYTHON_EXECUTABLE} ${CMAKE_CURRENT_SOURCE_DIR}/generate.py ${CMAKE_CURRENT_BINARY_DIR} WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})

find_program(DXC dxc REQUIRED)
message("Using dxc ${DXC}")

add_library(dxil-shaders STATIC ${CMAKE_CURRENT_BINARY_DIR}/shaders.hpp ${CMAKE_CURRENT_BINARY_DIR}/shaders.cpp)
target_link_libraries(dxil-shaders PUBLIC d3d12-device)
target_include_directories(dxil-shaders PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

function(compile_hlsl out_path in_path)
    add_custom_command(
        OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/${out_path}.inc
        DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/${in_path}.hlsl
        COMMAND ${DXC} -Tcs_6_5 -E ${out_path} -Fh ${CMAKE_CURRENT_BINARY_DIR}/${out_path}.inc -enable-16bit-types -force-rootsig-ver rootsig_1_0 ${CMAKE_CURRENT_SOURCE_DIR}/${in_path}.hlsl)
    add_custom_target(dxil-${out_path} DEPENDS ${CMAKE_CURRENT_BINARY_DIR}/${out_path}.inc)
    add_dependencies(dxil-shaders dxil-${out_path})
endfunction()
